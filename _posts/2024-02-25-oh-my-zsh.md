```
  cat /etc/shells


  chsh -s /usr/bin/zsh

  [reboot]
```

oh-my-zsh 를 curl 로 설치

```
sh -c "$(curl -fsSL https://raw.githubusercontent.com/ohmyzsh/ohmyzsh/master/tools/install.sh)"


vim ~/.zshrc


// 테마변경
ZSH_THEME="agnoster"
```
